## Endpoint

<CodeGroup>

```bash Requisição
POST /iptv
Authorization: Bearer <token>
```

</CodeGroup>

## Descrição

Cria um novo usuário ou teste IPTV no sistema.

## Parâmetros

<ParamField query="username" type="string" required>
  Nome de usuário para autenticação
</ParamField>

<ParamField query="password" type="string" required>
  Senha para autenticação
</ParamField>

<ParamField query="token" type="string" required>
  Token de autenticação
</ParamField>

<ParamField body="userline" type="string" required>
  Nome de usuário IPTV
</ParamField>

<ParamField body="passline" type="string" required>
  Senha do usuário
</ParamField>

<ParamField body="screen" type="integer">
  Número máximo de conexões simultâneas (padrão: 1)
</ParamField>

<ParamField body="isTrial" type="boolean">
  Se é um teste gratuito (padrão: false)
</ParamField>

<ParamField body="month" type="number">
  Duração em meses
</ParamField>

<ParamField body="packageId" type="string" required>
  ID do pacote IPTV
</ParamField>

<ParamField body="amount" type="number" required>
  Valor do pacote
</ParamField>

<ParamField body="contact" type="string" required>
  Informações de contato
</ParamField>

<ParamField body="whatsapp" type="string">
  Número do WhatsApp
</ParamField>

<ParamField body="notes" type="string">
  Observações adicionais
</ParamField>

## Resposta

<ResponseField name="success" type="boolean">
  Indica se a operação foi bem-sucedida
</ResponseField>

<ResponseField name="message" type="string">
  Mensagem de confirmação
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="properties">
    <ResponseField name="id" type="integer">
      ID do usuário IPTV criado
    </ResponseField>
    <ResponseField name="username" type="string">
      Nome de usuário IPTV
    </ResponseField>
    <ResponseField name="max_connections" type="integer">
      Número máximo de conexões
    </ResponseField>
    <ResponseField name="status" type="string">
      Status do usuário
    </ResponseField>
    <ResponseField name="expires_at" type="string">
      Data de expiração
    </ResponseField>
    <ResponseField name="connection_url" type="string">
      URL de conexão IPTV
    </ResponseField>
    <ResponseField name="m3u_url" type="string">
      URL do arquivo M3U
    </ResponseField>
  </Expandable>
</ResponseField>
