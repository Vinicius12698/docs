## Endpoint

<CodeGroup>

```bash Requisição
POST /iptv
Authorization: Bearer <token>
```

</CodeGroup>

## Descrição

Cria um novo usuário ou teste IPTV no sistema.

## Parâmetros

<ParamField header="Authorization" type="string" required>
  Bearer token para autenticação
</ParamField>

<ParamField body="username" type="string" required>
  Nome de usuário IPTV
</ParamField>

<ParamField body="password" type="string" required>
  Senha do usuário
</ParamField>

<ParamField body="max_connections" type="integer">
  Número máximo de conexões simultâneas (padrão: 1)
</ParamField>

<ParamField body="is_trial" type="boolean">
  Se é um teste gratuito (padrão: false)
</ParamField>

<ParamField body="trial_duration" type="integer">
  Duração do teste em horas (obrigatório se is_trial=true)
</ParamField>

<ParamField body="package_id" type="integer" required>
  ID do pacote IPTV
</ParamField>

<ParamField body="resale_id" type="integer" required>
  ID da revenda
</ParamField>

## Resposta

<ResponseField name="success" type="boolean">
  Indica se a operação foi bem-sucedida
</ResponseField>

<ResponseField name="message" type="string">
  Mensagem de confirmação
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="properties">
    <ResponseField name="id" type="integer">
      ID do usuário IPTV criado
    </ResponseField>
    <ResponseField name="username" type="string">
      Nome de usuário IPTV
    </ResponseField>
    <ResponseField name="max_connections" type="integer">
      Número máximo de conexões
    </ResponseField>
    <ResponseField name="status" type="string">
      Status do usuário
    </ResponseField>
    <ResponseField name="expires_at" type="string">
      Data de expiração
    </ResponseField>
    <ResponseField name="connection_url" type="string">
      URL de conexão IPTV
    </ResponseField>
    <ResponseField name="m3u_url" type="string">
      URL do arquivo M3U
    </ResponseField>
  </Expandable>
</ResponseField>
