## Endpoint

<CodeGroup>

```bash Requisição
POST /iptv/activate/app
Authorization: Bearer <token>
```

</CodeGroup>

## Descrição

Ativa um aplicativo IPTV específico para um usuário.

## Parâmetros

<ParamField header="Authorization" type="string" required>
  Bearer token para autenticação
</ParamField>

<ParamField body="user_id" type="integer" required>
  ID do usuário IPTV
</ParamField>

<ParamField body="app_name" type="string" required>
  Nome do aplicativo (ex: "smarttv", "mobile", "desktop")
</ParamField>

<ParamField body="device_id" type="string">
  ID único do dispositivo
</ParamField>

<ParamField body="device_name" type="string">
  Nome amigável do dispositivo
</ParamField>

## Resposta

<ResponseField name="success" type="boolean">
  Indica se a operação foi bem-sucedida
</ResponseField>

<ResponseField name="message" type="string">
  Mensagem de confirmação
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="properties">
    <ResponseField name="user_id" type="integer">
      ID do usuário IPTV
    </ResponseField>
    <ResponseField name="app_name" type="string">
      Nome do aplicativo ativado
    </ResponseField>
    <ResponseField name="device_id" type="string">
      ID do dispositivo
    </ResponseField>
    <ResponseField name="device_name" type="string">
      Nome do dispositivo
    </ResponseField>
    <ResponseField name="activation_code" type="string">
      Código de ativação
    </ResponseField>
    <ResponseField name="app_url" type="string">
      URL do aplicativo
    </ResponseField>
  </Expandable>
</ResponseField>
