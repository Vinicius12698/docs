## Endpoint

<CodeGroup>

```bash Requisição
PATCH /resale/add-credits/{id}
Authorization: Bearer <token>
```

</CodeGroup>

## Descrição

Adiciona ou remove créditos de uma revenda específica.

## Parâmetros

<ParamField path="id" type="integer" required>
  ID da revenda
</ParamField>

<ParamField query="username" type="string" required>
  Nome de usuário para autenticação
</ParamField>

<ParamField query="password" type="string" required>
  Senha para autenticação
</ParamField>

<ParamField query="token" type="string" required>
  Token de autenticação
</ParamField>

## Parâmetros do Body

<ParamField body="credits" type="string" required>
  Quantidade de créditos
</ParamField>

<ParamField body="reason" type="string">
  Motivo da alteração
</ParamField>

<ParamField body="sale" type="number">
  Valor de venda
</ParamField>

## Resposta

<ResponseField name="success" type="boolean">
  Indica se a operação foi bem-sucedida
</ResponseField>

<ResponseField name="message" type="string">
  Mensagem de confirmação
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="properties">
    <ResponseField name="id" type="integer">
      ID da revenda
    </ResponseField>
    <ResponseField name="previous_credits" type="number">
      Créditos anteriores
    </ResponseField>
    <ResponseField name="new_credits" type="number">
      Novos créditos
    </ResponseField>
    <ResponseField name="credits_added" type="number">
      Créditos adicionados/removidos
    </ResponseField>
    <ResponseField name="reason" type="string">
      Motivo da alteração
    </ResponseField>
  </Expandable>
</ResponseField>
