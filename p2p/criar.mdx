## Endpoint

<CodeGroup>

```bash Requisição
POST /p2p
Authorization: Bearer <token>
```

</CodeGroup>

## Descrição

Cria um novo usuário ou teste P2P no sistema.

## Parâmetros

<ParamField query="username" type="string" required>
  Nome de usuário para autenticação
</ParamField>

<ParamField query="password" type="string" required>
  Senha para autenticação
</ParamField>

<ParamField query="token" type="string" required>
  Token de autenticação
</ParamField>

<ParamField body="amount" type="integer" required>
  Quantidade/valor
</ParamField>

<ParamField body="contact" type="string" required>
  Contato do usuário
</ParamField>

<ParamField body="isTrial" type="boolean" required>
  Se é um teste gratuito
</ParamField>

<ParamField body="month" type="number" required>
  Duração em meses
</ParamField>

<ParamField body="notes" type="string">
  Observações
</ParamField>

<ParamField body="packageId" type="string" required>
  ID do pacote P2P
</ParamField>

<ParamField body="passline" type="string" required>
  Senha de linha
</ParamField>

<ParamField body="sale_value" type="number" required>
  Valor da venda
</ParamField>

<ParamField body="typeUser" type="integer" required>
  Tipo do usuário
</ParamField>

<ParamField body="userline" type="string" required>
  Linha do usuário
</ParamField>

<ParamField body="whatsapp" type="number" required>
  WhatsApp do usuário (apenas números)
</ParamField>

## Resposta

<ResponseField name="success" type="boolean">
  Indica se a operação foi bem-sucedida
</ResponseField>

<ResponseField name="message" type="string">
  Mensagem de confirmação
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="properties">
    <ResponseField name="id" type="integer">
      ID do usuário P2P criado
    </ResponseField>
    <ResponseField name="username" type="string">
      Nome de usuário P2P
    </ResponseField>
    <ResponseField name="status" type="string">
      Status do usuário
    </ResponseField>
    <ResponseField name="max_devices" type="integer">
      Número máximo de dispositivos
    </ResponseField>
    <ResponseField name="expires_at" type="string">
      Data de expiração
    </ResponseField>
    <ResponseField name="server_url" type="string">
      URL do servidor P2P
    </ResponseField>
    <ResponseField name="streaming_key" type="string">
      Chave de streaming
    </ResponseField>
  </Expandable>
</ResponseField>
