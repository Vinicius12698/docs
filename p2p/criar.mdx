## Endpoint

<CodeGroup>

```bash Requisição
POST /p2p
```

</CodeGroup>

## Descrição

Cria um novo usuário ou teste P2P no sistema.

## Parâmetros

<ParamField body="username" type="string" required>
  Nome de usuário P2P
</ParamField>

<ParamField body="password" type="string" required>
  Senha do usuário
</ParamField>

<ParamField body="is_trial" type="boolean">
  Se é um teste gratuito (padrão: false)
</ParamField>

<ParamField body="trial_duration" type="integer">
  Duração do teste em horas (obrigatório se is_trial=true)
</ParamField>

<ParamField body="package_id" type="integer" required>
  ID do pacote P2P
</ParamField>

<ParamField body="resale_id" type="integer" required>
  ID da revenda
</ParamField>

<ParamField body="max_devices" type="integer">
  Número máximo de dispositivos (padrão: 1)
</ParamField>

## Resposta

<ResponseField name="success" type="boolean">
  Indica se a operação foi bem-sucedida
</ResponseField>

<ResponseField name="message" type="string">
  Mensagem de confirmação
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="properties">
    <ResponseField name="id" type="integer">
      ID do usuário P2P criado
    </ResponseField>
    <ResponseField name="username" type="string">
      Nome de usuário P2P
    </ResponseField>
    <ResponseField name="status" type="string">
      Status do usuário
    </ResponseField>
    <ResponseField name="max_devices" type="integer">
      Número máximo de dispositivos
    </ResponseField>
    <ResponseField name="expires_at" type="string">
      Data de expiração
    </ResponseField>
    <ResponseField name="server_url" type="string">
      URL do servidor P2P
    </ResponseField>
    <ResponseField name="streaming_key" type="string">
      Chave de streaming
    </ResponseField>
  </Expandable>
</ResponseField>
