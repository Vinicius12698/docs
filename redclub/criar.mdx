## Endpoint

<CodeGroup>

```bash Requisição
POST /red-club
```

</CodeGroup>

## Descrição

Cria um novo usuário RedClub no sistema.

## Parâmetros

<ParamField body="username" type="string" required>
  Nome de usuário RedClub
</ParamField>

<ParamField body="password" type="string" required>
  Senha do usuário
</ParamField>

<ParamField body="email" type="string" required>
  E-mail do usuário
</ParamField>

<ParamField body="phone" type="string">
  Telefone do usuário
</ParamField>

<ParamField body="is_trial" type="boolean">
  Se é um teste gratuito (padrão: false)
</ParamField>

<ParamField body="trial_duration" type="integer">
  Duração do teste em horas (obrigatório se is_trial=true)
</ParamField>

<ParamField body="package_id" type="integer" required>
  ID do pacote RedClub
</ParamField>

<ParamField body="resale_id" type="integer" required>
  ID da revenda
</ParamField>

<ParamField body="profile_type" type="string">
  Tipo de perfil (basic/premium/vip)
</ParamField>

## Resposta

<ResponseField name="success" type="boolean">
  Indica se a operação foi bem-sucedida
</ResponseField>

<ResponseField name="message" type="string">
  Mensagem de confirmação
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="properties">
    <ResponseField name="id" type="integer">
      ID do usuário RedClub criado
    </ResponseField>
    <ResponseField name="username" type="string">
      Nome de usuário RedClub
    </ResponseField>
    <ResponseField name="email" type="string">
      E-mail do usuário
    </ResponseField>
    <ResponseField name="phone" type="string">
      Telefone do usuário
    </ResponseField>
    <ResponseField name="status" type="string">
      Status do usuário
    </ResponseField>
    <ResponseField name="profile_type" type="string">
      Tipo de perfil
    </ResponseField>
    <ResponseField name="expires_at" type="string">
      Data de expiração
    </ResponseField>
    <ResponseField name="access_url" type="string">
      URL de acesso ao portal
    </ResponseField>
    <ResponseField name="member_id" type="string">
      ID de membro único
    </ResponseField>
  </Expandable>
</ResponseField>
