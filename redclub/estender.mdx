## Endpoint

<CodeGroup>

```bash Requisição
POST /red-club/extend/{id}
```

</CodeGroup>

## Descrição

Estende a validade de um usuário RedClub existente.

## Parâmetros de URL

<ParamField path="id" type="integer" required>
  ID do usuário RedClub
</ParamField>

## Parâmetros do Body

<ParamField body="extend_days" type="integer" required>
  Número de dias para estender
</ParamField>

<ParamField body="package_id" type="integer">
  Novo ID do pacote (opcional)
</ParamField>

<ParamField body="profile_type" type="string">
  Novo tipo de perfil (opcional)
</ParamField>

## Resposta

<ResponseField name="success" type="boolean">
  Indica se a operação foi bem-sucedida
</ResponseField>

<ResponseField name="message" type="string">
  Mensagem de confirmação
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="properties">
    <ResponseField name="id" type="integer">
      ID do usuário RedClub
    </ResponseField>
    <ResponseField name="username" type="string">
      Nome de usuário
    </ResponseField>
    <ResponseField name="previous_expires_at" type="string">
      Data de expiração anterior
    </ResponseField>
    <ResponseField name="new_expires_at" type="string">
      Nova data de expiração
    </ResponseField>
    <ResponseField name="days_extended" type="integer">
      Dias estendidos
    </ResponseField>
    <ResponseField name="profile_type" type="string">
      Novo tipo de perfil
    </ResponseField>
    <ResponseField name="member_id" type="string">
      ID de membro atualizado
    </ResponseField>
  </Expandable>
</ResponseField>
